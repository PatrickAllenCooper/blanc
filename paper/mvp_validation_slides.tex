\section{Motivation}
% ==============================================================================

\begin{frame}{The Challenge: Three Entangled Deficits}

Foundation models are routinely evaluated on forward, monotonic inference.
Scientific progress demands something harder.

\vspace{0.8em}
\begin{columns}[T]
\begin{column}{0.33\textwidth}
\centering
\textbf{\textcolor{defabblue}{Grounding Deficit}}\\[4pt]
\small
Models cannot trace conclusions to supporting evidence.\\[4pt]
\textit{Root cause of hallucination.}
\end{column}

\begin{column}{0.33\textwidth}
\centering
\textbf{\textcolor{defabblue}{Novelty Deficit}}\\[4pt]
\small
Models cannot identify which beliefs are revisable or where exceptions apply.\\[4pt]
\textit{Root cause of creative sterility.}
\end{column}

\begin{column}{0.33\textwidth}
\centering
\textbf{\textcolor{defabblue}{Belief Revision Deficit}}\\[4pt]
\small
Models cannot update knowledge while preserving unrelated commitments.\\[4pt]
\textit{Root cause of knowledge editing fragility.}
\end{column}
\end{columns}

\vspace{1em}
These deficits are \textbf{mutually reinforcing}: without grounding, no targeted revision; without revision, no genuine creativity; without novelty, no revision at all.
\end{frame}


\begin{frame}{The Tautological Bottleneck}

\begin{quote}
The very hypotheses a model would need to generate are precisely those it has never seen, because the act of proposing them is what converts them from creative conjectures into textbook facts.
\end{quote}

\vspace{0.8em}
\textbf{Example}: Intrinsically disordered proteins (IDPs)

\begin{itemize}
    \item Prior dogma: proteins require stable 3D structure to function
    \item Discovery: proteins without fixed structure remain functionally essential
    \item An AI trained on pre-discovery data would resist this hypothesis---it directly contradicts the default encoded in its training corpus
\end{itemize}

\vspace{0.8em}
The creative act requires not merely asserting the exception, but doing so via a \textbf{targeted revision} that overrides one default while leaving all other predictions intact.
\end{frame}


\begin{frame}{Defeasible Reasoning as the Missing Structure}

\small
\textbf{Defeasible logic} provides exactly what foundation models lack:

\vspace{0.2em}
\footnotesize
\begin{description}[labelwidth=2.1cm, leftmargin=2.3cm]
    \item[Grounding] Every conclusion has a traceable support set
    \item[Revisability] Every default is explicitly marked as overridable
    \item[Conservativity] Revisions preserve all unrelated expectations (AGM minimal change)
\end{description}

\small
\vspace{0.2em}
\textbf{A defeasible theory} $\D = (F, R_s, R_d, R_{df}, {\succ})$:
\begin{itemize}\setlength\itemsep{0pt}
    \item \textbf{Facts} $F$ -- ground observations
    \item \textbf{Strict rules} $R_s$ -- invariable ($A \strictto B$)
    \item \textbf{Defeasible rules} $R_d$ -- revisable defaults ($A \defto B$)
    \item \textbf{Defeaters} $R_{df}$ -- block but cannot support ($A \defeatto \neg B$)
    \item \textbf{Superiority} $\succ$ -- acyclic priority over conflicting rules
\end{itemize}
\end{frame}

\begin{frame}{Three Levels of Increasing Difficulty}

\begin{columns}[T]
\begin{column}{0.32\textwidth}
\textbf{\textcolor{defabgreen}{Level 1: Fact Completion}}\\[4pt]
\small
Ablate a ground fact $e \in F$.\\[4pt]
Task: identify the missing observation.\\[6pt]
\textit{``Which bird is aquatic?''}\\[6pt]
Tests: \textbf{Grounding}
\end{column}

\begin{column}{0.32\textwidth}
\textbf{\textcolor{defabgold}{Level 2: Rule Abduction}}\\[4pt]
\small
Ablate a defeasible rule $e \in R_d$.\\[4pt]
Task: reconstruct the missing generalization.\\[6pt]
\textit{``What rule predicts flight?''}\\[6pt]
Tests: \textbf{Grounding + Novelty}
\end{column}

\begin{column}{0.32\textwidth}
\textbf{\textcolor{defabredish}{Level 3: Defeater Abduction}}\\[4pt]
\small
Construct a novel defeater that resolves an anomaly conservatively.\\[6pt]
\textit{``Why doesn't this penguin fly?''}\\[6pt]
Tests: \textbf{Grounding + Novelty + Revision}
\end{column}
\end{columns}

\vspace{1em}
\begin{center}
\small
Progression: \textbf{Observation} $\longrightarrow$ \textbf{Generalization} $\longrightarrow$ \textbf{Conservative Exception}\\[4pt]
The formal analogue of rational scientific revision.
\end{center}
\end{frame}

\section{Formal Foundations}
% ==============================================================================

\begin{frame}{Defeasible Conversion and Criticality}

\small
\textbf{Conversion}: $\partfunc: \Pi \to \{s, d\}$ classifies each clause as strict or defeasible:
\vspace{-0.3em}
\[
\phi_\partfunc(\Pi) = (F_\partfunc,\; R_s^{\partfunc},\; R_d^{\partfunc},\; \emptyset,\; \emptyset)
\]

\vspace{-0.5em}
\textbf{Five families}: $\partfunc_{\mathrm{leaf}}$, $\partfunc_{\mathrm{rule}}$, $\partfunc_{\mathrm{depth}}(k)$, $\partfunc_{\mathrm{rand}}(\delta)$, $\partfunc_{\mathrm{type}}$.

\vspace{0.2em}
\textbf{Criticality}: Element $e$ is critical for $q$ if $\D \setminus \{e\} \not\pPartial q$.

\vspace{0.2em}
\footnotesize
\begin{center}
\begin{tabular}{@{}lc@{}}
\toprule
Operation & Complexity \\
\midrule
Defeasible derivation $\D \pPartial q$ & $O(|R| \cdot |F|)$ \\
Full-theory criticality $\CritStar(\D, q)$ & $O(|\D|^2 \cdot |F|)$ \\
Full pipeline per instance & $O(|\D|^3)$ \\
Weak DAP resolution & NP-complete \\
Strong DAP resolution & $\Sigma_2^P$-complete \\
\bottomrule
\end{tabular}
\end{center}

\small
\vspace{0.2em}
Instance generation uses $\CritStar$ (polynomial), not full support enumeration (NP-hard).
\end{frame}


\begin{frame}{Level 3: Conservativity as AGM Minimal Change}

\small
\textbf{Anomaly}: $\alpha$ is anomalous when $\D \pPartial \neg\alpha$ but only via defeasible rules.

\textbf{Resolution}: A defeater $r$ such that $\D \cup \{r\} \not\pPartial \neg\alpha$.

\textbf{Conservativity}: Preserves all expectations except $\neg\alpha$:
\vspace{-0.3em}
\[
\forall q \in \Exp(\D) \setminus \{\neg\alpha\}: \quad \D \cup \{r\} \pPartial q
\]

\vspace{-0.3em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Resolution strength}:
\begin{itemize}\setlength\itemsep{1pt}
    \item \textbf{Weak}: theory becomes agnostic
    \item \textbf{Strong}: $\alpha$ becomes provable
    \item \textbf{Restruct.}: priority reorganized
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Creativity measures}:
\begin{itemize}\setlength\itemsep{1pt}
    \item $\Nov(r, \Dm)$: fraction of novel predicates
    \item $d_{\mathrm{rev}}$: revision distance (AGM analogue)
\end{itemize}
\end{column}
\end{columns}

\vspace{0.3em}
Resolving the anomaly but disrupting other predictions scores 0.5 --- lacking the \textbf{rational revision} that AGM minimal change requires.
\end{frame}

\begin{frame}{Key Takeaways}

\small
\begin{enumerate}\setlength\itemsep{1pt}
    \item \textbf{Grounding is largely solved}: 73--79\% L2 across formal modalities.
    \item \textbf{Belief revision is latent, not absent}: reasoning models reach 65--93\% L3 under CoT; instruction models peak at 24\%.
    \item \textbf{Narrative rendering is the universal bottleneck}: 55--70 pp gap between M1 and formal modalities.
    \item \textbf{CoT helps generation, hurts selection}: ``overthinking'' at L2 vs.\ latent reasoning activation at L3.
    \item \textbf{Formal verification enables exact fine-tuning}: polynomial-time verifier as reward signal with no approximation error.
\end{enumerate}

\textbf{DeFAb measures not just whether a model can be creative,\\but whether it can be \textit{rationally} creative.}
