\section{Motivation: Why Defeasible Logic for ROE?}
% =============================================================================

\begin{frame}{The Rules of Engagement Problem}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{What ROE actually are:}
      \begin{itemize}[leftmargin=1.2em]
        \item A layered system of \emph{defaults and exceptions}
        \item Self-defense \textbf{overrides} mission constraints
        \item Mission constraints \textbf{override} general policy
        \item Newport Handbook: ``limitations on mission
              accomplishment have no impact on the commander's
              inherent right\ldots{}to exercise self-defense''
      \end{itemize}

      \vspace{0.6em}
      \textbf{The formal match:}\\[0.3em]
      \textit{[diagram]}
    \end{column}

    \begin{column}{0.44\textwidth}
      \textbf{Why this is hard for LLMs:}
      \vspace{0.3em}
      \begin{enumerate}[leftmargin=1.4em, label=\arabic*.]
        \item Rules \emph{conflict} in edge cases
        \item Exceptions must be \emph{conservative} (defeat only the
              targeted constraint, not the whole policy)
        \item Correctness is formally verifiable in \textbf{polynomial time}
      \end{enumerate}

      \vspace{0.7em}
      \alert{Key insight:} a polynomial-time verifier can be the
      ground-truth judge of LLM compliance --- no human annotation
      required.
    \end{column}
  \end{columns}
\end{frame}

\section{The Defeasible Theory Structure}
% =============================================================================

\begin{frame}{ROE as a Defeasible Theory: Worked Example}

  \textbf{Scenario: Marine in restricted exclusion zone, enemy approaches}

  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \textbf{Theory $\D$ (full, with defeater):}\\[0.3em]
      \small
      \textit{Strict:}\\
      \texttt{military\_unit(X) $\to$ unit(X)}\\[2pt]
      \textit{Default ROE rule (rts\_r3000):}\\
      \texttt{military\_unit(X), military\_target(Y)}\\
      $\;\Rightarrow$ \texttt{authorized\_to\_engage(X,Y)}\\[2pt]
      \textit{Defeater (rts\_r3003):}\\
      \texttt{in\_zone(X, restricted), in\_zone(Y, restricted)}\\
      $\;\leadsto$ $\neg$\texttt{authorized\_to\_engage(X,Y)}\\[2pt]
      \textit{Superiority:} rts\_r3007 $\succ$ rts\_r3003\\[4pt]
      \textbf{Result:} $\D \not\vdash_\partial$ \texttt{authorized\_to\_engage}\\
      Engagement \textbf{prohibited}. \roecheck
    \end{column}

    \begin{column}{0.46\textwidth}
      \textbf{Theory $\D^-$ (defeater removed):}\\[0.3em]
      \small
      Same strict and defeasible rules,\\
      \textbf{defeater rts\_r3003 absent.}\\[0.5em]
      \textbf{Result:}\\
      $\D^- \vdash_\partial$ \texttt{authorized\_to\_engage(marine, enemy)}\\[0.3em]
      \alert{Anomaly}: engagement authorized in exclusion zone!\\[0.6em]

      \textbf{L3 task:} given $\D^-$ and the anomaly,\\
      construct the missing defeater conservatively.\\[0.4em]
      Gold answer: rts\_r3003\\
      \texttt{in\_zone(X,restricted), in\_zone(Y,restricted)}\\
      $\leadsto$ $\neg$\texttt{authorized\_to\_engage(X,Y)}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The Worker Protection Asymmetry}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Why does S2 (worker protection) trigger violations\\
      while S1/S3/S4 never do?}

      \vspace{0.5em}
      Models have strong \emph{spatial} ROE intuitions:\\
      zone boundaries are respected even without explicit instruction.

      \vspace{0.4em}
      But \emph{object-level protection} is weaker:\\
      under ``kill everything'' pressure, models propose
      attacking the worker --- an order that the verifier
      correctly identifies and rejects.

      \vspace{0.4em}
      \textbf{Interpretation:}\\
      LLMs encode spatial norms implicitly\\
      (possibly from game-play corpora),\\
      but object-level \emph{non-combatant} status
      requires explicit ROE scaffolding.
    \end{column}

\section{Main Claims}

\begin{frame}{What We Can Now Assert}

  \begin{enumerate}[leftmargin=1.5em, itemsep=6pt]
    \item \textbf{ROE formalizes correctly as a defeasible theory.}\\
          KB certified: 148 rules, 27 defeaters, 15 superiority relations,
          6 seed conflicts all pass the triple (anomaly / block / conservativity).

    \item \textbf{LLM performance on ROE is discriminative and model-specific.}\\
          CoT gap: nano 17\%$\to$100\%; Claude dissociation (83\% direct, 0\% CoT)
          replicates on disjoint domain. These are structural differences,
          not vocabulary artefacts.

    \item \textbf{H4 refuted: ROE syntax is harder, not easier.}\\
          L3 decoder failure rate is 61.8\% (vs.\ 38.1\% overall at L2).
          The \texttt{:$\leadsto$} defeater notation is a genuine output challenge.

    \item \textbf{B2 gate recovery works: 100\% success rate.}\\
          Aggressive commander prompt triggers worker-protection violations.
          Verifier catches every proposed violation; model revises to compliant
          order within 0.7--2.8 reprompts. Gate never admitted a violating order.

    \item \textbf{Object-level protection is weaker than spatial ROE alignment.}\\
          Models never violate zone-based rules (exclusion zone, civilian area)
          even under pure-aggressive prompting. Worker protection breaks.
          This identifies where formal ROE scaffolding is most needed.
  \end{enumerate}
\end{frame}

\begin{frame}{Appendix: ROE Theory Formally}
  \small
  A defeasible theory $\D = (F, R_s, R_d, R_{df}, {\succ})$:

  \begin{itemize}[leftmargin=1.2em, itemsep=3pt]
    \item $F$: ground facts (lifted SC2 game state each tick)
    \item $R_s$: strict rules (unit taxonomy, weapon capabilities --- game physics)
    \item $R_d$: defeasible rules (ROE defaults: authorize, protect, retreat, stealth)
    \item $R_{df}$: defeaters (ROE exceptions: self-defense, escalation, override)
    \item ${\succ}$: superiority (self-defense $\succ$ exclusion zone prohibition)
  \end{itemize}

  \vspace{0.4em}
  \textbf{Derivability ($O(|R|^2)$ per Theorem 11):}\\
  $+\partial q$ holds iff (i)~$+\Delta q$ (strictly provable), OR\\
  \phantom{$+\partial q$ holds iff }
  (ii)~$-\Delta\overline{q}$ AND some defeasible rule fires for $q$ AND
  all attackers are either inapplicable or defeated by a superior rule.

  \vspace{0.4em}
  \textbf{Conservativity of defeater $d$:}\\
  $|\mathrm{Exp}(\D \cup \{d\}) \setminus \mathrm{Exp}(\D)| +
   |\mathrm{Exp}(\D) \setminus \mathrm{Exp}(\D \cup \{d\})| = 0$
  except for the targeted anomaly literal $\overline{q}$.\\
  This is the formal counterpart of the Newport proportionality principle.
\end{frame}
